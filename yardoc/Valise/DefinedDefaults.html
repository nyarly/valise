def add_item(path, contents)
  check_path = path[0..-2]
  until check_path.empty?
    if @files.has_key?(check_path)
      raise MalformedTree, "Tried to add items below #{path[0..-2]} which is not a directory"
    end
    check_path.pop
  end
  @files[path] = contents
end

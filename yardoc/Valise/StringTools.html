def align(string)
  lines = string.split(/\n/)
  first = lines.shift
  match = /^(\s*)<<</.match(first)
  unless(match.nil?)
    return lines.map do |line|
      unless /^#{match[1]}|^\s*$/ =~ line
        raise Errors::UnderIndented, line
      end
      line.sub(/^#{match[1]}/, "")
    end.join("\n")
  end
  return string
end

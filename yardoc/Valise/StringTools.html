def align(string)
  lines = string.split(/\n/)
  first = lines.shift
  match = /^(\s*)<<</.match(first)
  unless(match.nil?)
    catch(:haircut) do
      return lines.map do |line|
        raise line if /^#{match[1]}|^\s*$/ !~ line
        throw :haircut if /^#{match[1]}|^\s*$/ !~ line
        line.sub(/^#{match[1]}/, "")
      end.join("\n")
    end
  end
  return string
end

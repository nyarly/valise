def each
  return enum_for(:each) unless block_given?
  @stack.each do |item|
    @extensions.each do |ext|
      dir = item.segments.dup
      file = dir.pop
      ext_stack = @stacks[dir + [file + ext]]
      yield(ext_stack.item_for(item.root))
    end
  end
end
